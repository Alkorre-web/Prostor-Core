# Конвертация Row → dict
def row_to_dict(row):
    return dict(row) if row else None

def rows_to_dicts(rows):
    return [dict(row) for row in rows]

# Использование

    'SELECT * FROM skills WHERE user_id = ? LIMIT 4',
